class Solution {
    public boolean isValid(String s) {
        if (s.length() % 2 == 1)
			return false;

Stack<Character> stack = new Stack<>();
		char c[] = s.toCharArray();
		for (int i = 0; i < c.length; i++) {
			if (c[i] == '(' || c[i] == '[' || c[i] == '{') {
				stack.push(c[i]);
			} else if (c[i] == ')' && !stack.isEmpty() && stack.peek() == '(') {
				stack.pop();
			} else if (c[i] == '}' && !stack.isEmpty() && stack.peek() == '{') {
				stack.pop();
			} else if (c[i] == ']' && !stack.isEmpty() && stack.peek() == '[') {
				stack.pop();
			} else
				return false;
		}
		return stack.isEmpty();  
    }
}
